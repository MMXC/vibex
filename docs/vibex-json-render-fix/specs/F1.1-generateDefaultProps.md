## 修复方案

在 `canvasApi.ts` 新增 `generateDefaultProps()` 函数：

```typescript
function generateDefaultProps(type: string, name: string): Record<string, unknown> {
  switch (type) {
    case 'page':
      return { title: name, layout: 'topnav' };
    case 'form':
      return {
        title: name,
        fields: [
          { name: 'email', label: '邮箱', type: 'text', placeholder: '请输入邮箱', required: true },
          { name: 'password', label: '密码', type: 'password', placeholder: '请输入密码', required: true },
        ],
        submitLabel: '提交',
      };
    case 'list':
      return {
        title: name,
        columns: [
          { key: 'id', label: 'ID', sortable: false },
          { key: 'name', label: '名称', sortable: true },
        ],
        rows: 10,
        searchable: true,
      };
    case 'detail':
      return {
        title: name,
        fields: [
          { label: '状态', value: '待处理' },
          { label: '创建时间', value: new Date().toISOString().split('T')[0] },
        ],
      };
    case 'modal':
      return { title: name, size: 'md' };
    default:
      return { title: name };
  }
}
```

**要求**: 字段结构严格参考 `catalog.ts` 中的 Zod schema。

## 验收标准

```typescript
// page 类型
expect(generateDefaultProps('page', 'MyPage')).toMatchObject({ title: 'MyPage', layout: 'topnav' });

// form 类型
const formProps = generateDefaultProps('form', 'LoginForm');
expect(formProps).toMatchObject({ title: 'LoginForm', fields: expect.any(Array) });
expect((formProps as any).fields.length).toBeGreaterThanOrEqual(2);

// list 类型
const listProps = generateDefaultProps('list', 'UserTable');
expect(listProps).toMatchObject({ title: 'UserTable', columns: expect.any(Array), rows: 10, searchable: true });

// detail 类型
const detailProps = generateDefaultProps('detail', 'UserDetail');
expect(detailProps).toMatchObject({ title: 'UserDetail', fields: expect.any(Array) });

// modal 类型
expect(generateDefaultProps('modal', 'ConfirmDialog')).toMatchObject({ title: 'ConfirmDialog', size: 'md' });

// unknown 类型兜底
expect(generateDefaultProps('unknown', 'X')).toMatchObject({ title: 'X' });
```
